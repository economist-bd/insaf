// কনফিগারেশন const itemsPerPage = 20; let currentPage = 1; // পেজ লোড ফাংশন function loadTips(page) { const container = document.getElementById('tipsContainer'); container.innerHTML = ''; // আগের কন্টেন্ট পরিষ্কার করা // শুরু এবং শেষের ইনডেক্স বের করা const startIndex = (page - 1) * itemsPerPage; const endIndex = startIndex + itemsPerPage; // ডাটা স্লাইস করা const currentTips = tipsData.slice(startIndex, endIndex); // কার্ড জেনারেট করা currentTips.forEach(tip => { const cardHTML = `
${tip.title}

${tip.text}

`; container.innerHTML += cardHTML; }); updateButtons(); } // বাটন ও পেজ নম্বর আপডেট করা function updateButtons() { const totalPages = Math.ceil(tipsData.length / itemsPerPage); // টেক্সট আপডেট document.getElementById('currentPageNum').innerText = currentPage; document.getElementById('totalPagesNum').innerText = totalPages; // বাটন এনাবল/ডিজেবল লজিক document.getElementById('prevBtn').disabled = (currentPage === 1); document.getElementById('nextBtn').disabled = (currentPage === totalPages); } // পেজ পরিবর্তনের ফাংশন function changePage(direction) { currentPage += direction; loadTips(currentPage); // পেজ চেঞ্জ হলে উপরে স্ক্রল করবে document.getElementById('tipsContainer').scrollIntoView({ behavior: 'smooth' }); } // শেয়ার ফাংশন (Web Share API) function shareTip(title, text) { if (navigator.share) { navigator.share({ title: 'ইনসাফ মেডিকেল হেলথ টিপস', text: `${title}: ${text}`, url: window.location.href }) .catch((error) => console.log('Error sharing', error)); } else { // পিসির জন্য বা যাদের শেয়ার API নেই alert('টিপসটি ক্লিপবোর্ডে কপি করা হয়েছে: ' + title); navigator.clipboard.writeText(`${title}: ${text} - ইনসাফ মেডিকেল`); } } // শুরুতে প্রথম পেজ লোড হবে window.onload = function() { loadTips(1); };
